###############################################
# RDS MySQL — DEVELOPMENT / NON‑PRODUCTION
###############################################
# This instance is optimized for dev/test environments:
# - Low‑cost instance class
# - Minimal backups
# - No Multi‑AZ
# - No final snapshot
# - Fast teardown
# - Credentials sourced from Secrets Manager JSON
resource "aws_db_instance" "catalog_rds" {
  identifier               = "mydb3"
  engine                   = "mysql"
  engine_version           = "8.0"
  instance_class           = "db.t3.micro"
  allocated_storage        = 20
  db_name                  = "catalogdb"
  username                 = local.app_store_secret_json.username
  password                 = local.app_store_secret_json.password

  db_subnet_group_name     = aws_db_subnet_group.rds_private.name
  vpc_security_group_ids   = [aws_security_group.rds_mysql_sg.id]

  publicly_accessible      = false
  skip_final_snapshot      = true
  delete_automated_backups = true
  multi_az                 = false
  backup_retention_period  = 1

  tags = {
    Name = "${local.name_prefix}-catalog-rds-mysql"
    Environment = "dev"
  }
}

output "catalog_rds_endpoint" {
  description = "RDS endpoint for Catalog microservice"
  value       = aws_db_instance.catalog_rds.address
}

output "catalog_rds_sg_id" {
  description = "RDS security group ID"
  value       = aws_security_group.rds_mysql_sg.id
}



###############################################################
# RDS MySQL — PRODUCTION‑GRADE CONFIGURATION (Recommended)
###############################################################
# This block represents a production‑ready RDS MySQL instance.
# Key production features:
# - Multi‑AZ for high availability
# - Storage autoscaling to prevent outages
# - KMS encryption using CMK
# - Performance Insights enabled
# - Enhanced Monitoring enabled
# - Backup retention 7–30 days
# - Final snapshot on deletion
# - Parameter group for MySQL tuning
# - No public access
# - Credentials sourced from Secrets Manager JSON
resource "aws_db_instance" "catalog_rds_prod" {
  identifier               = "${local.name_prefix}-catalog-rds-prod"
  engine                   = "mysql"
  engine_version           = "8.0"
  instance_class           = "db.m6g.large" # Production‑grade Graviton instance

  # Storage
  allocated_storage        = 50
  max_allocated_storage    = 200 # Autoscaling to avoid storage‑full failures
  storage_type             = "gp3"
  storage_encrypted        = true
  kms_key_id               = aws_kms_key.rds_kms_key.arn

  # Initial DB
  db_name                  = "catalogdb"
  username                 = local.app_store_secret_json.username
  password                 = local.app_store_secret_json.password

  # Networking
  db_subnet_group_name     = aws_db_subnet_group.rds_private.name
  vpc_security_group_ids   = [aws_security_group.rds_mysql_sg.id]
  publicly_accessible      = false

  # High Availability
  multi_az                 = true

  # Backups & Snapshots
  backup_retention_period  = 14
  preferred_backup_window  = "03:00-04:00"
  skip_final_snapshot      = false
  final_snapshot_identifier = "${local.name_prefix}-catalog-rds-final"

  # Monitoring
  monitoring_interval      = 30
  monitoring_role_arn      = aws_iam_role.rds_monitoring_role.arn
  performance_insights_enabled = true
  performance_insights_retention_period = 7

  # Maintenance
  auto_minor_version_upgrade = true
  apply_immediately          = false
  preferred_maintenance_window = "sun:04:00-sun:05:00"

  # Parameter group for MySQL tuning
  parameter_group_name = aws_db_parameter_group.mysql_prod_params.name

  tags = {
    Name        = "${local.name_prefix}-catalog-rds-mysql-prod"
    Environment = "production"
    Owner       = "platform-team"
  }
}

###############################################################
# Supporting Resources for Production RDS
###############################################################

# KMS Key for RDS Encryption
resource "aws_kms_key" "rds_kms_key" {
  description = "KMS CMK for RDS MySQL encryption"
  enable_key_rotation = true
}

# Enhanced Monitoring IAM Role
resource "aws_iam_role" "rds_monitoring_role" {
  name = "${local.name_prefix}-rds-monitoring-role"

  assume_role_policy = jsonencode({
    Version = "2012-10-17",
    Statement = [{
      Effect = "Allow",
      Principal = { Service = "monitoring.rds.amazonaws.com" },
      Action = "sts:AssumeRole"
    }]
  })
}

resource "aws_iam_role_policy_attachment" "rds_monitoring_policy" {
  role       = aws_iam_role.rds_monitoring_role.name
  policy_arn = "arn:aws:iam::aws:policy/service-role/AmazonRDSEnhancedMonitoringRole"
}

# MySQL Parameter Group
resource "aws_db_parameter_group" "mysql_prod_params" {
  name   = "${local.name_prefix}-mysql-prod-params"
  family = "mysql8.0"

  # Example tuning
  parameter {
    name  = "max_connections"
    value = "500"
  }

  parameter {
    name  = "slow_query_log"
    value = "1"
  }

  parameter {
    name  = "long_query_time"
    value = "2"
  }

  tags = {
    Environment = "production"
  }
}

# Output for Production RDS
output "catalog_rds_prod_endpoint" {
  description = "Production RDS endpoint for Catalog microservice"
  value       = aws_db_instance.catalog_rds_prod.address
}
